            p: @COMP_CLASS@

    override protected fun getTags(p: @COMP_CLASS@): Set<String>  = HashSet()

    override protected fun isTemplateSuitable(
            p: @COMP_CLASS@

    override public fun getTemplateFromElement(newP: @COMP_CLASS@): PsiTemplateGen<@COMP_CLASS@, SmartInsertPlace>? = null
